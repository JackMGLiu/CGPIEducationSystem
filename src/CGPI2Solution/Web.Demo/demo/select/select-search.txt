                                                                text = $el.data('text');
                                                            if (othis.hasClass(THIS)) {

                                                                    value === '' || (origin === 'blur')

                                                            });

                                                            keyCode === 38 ||
                                                            keyCode === 39 ||
                                                            keyCode === 40

                                                            return false;

                                                        notOption(value,

                                                                        $dl.append('<p class="' + NONE + '">无匹配项</p>');
                                                                } else {
                                                                    $dl.find('.' + NONE).remove();
                                                                }
                                                            },

                                                            $dl.find('.' + NONE).remove();
